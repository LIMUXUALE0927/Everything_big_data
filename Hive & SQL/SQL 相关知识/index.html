    SQL зӣёе…ізҹҘиҜҶ

    MySQL иҜӯеҸҘжү§иЎҢйЎәеәҸ

    жҹҘиҜўиҜӯеҸҘзҡ„жү§иЎҢйЎәеәҸ:

    (7) SELECT
(8) DISTINCT <select_list>
(1) FROM <left_table>
(2) <join_type> JOIN <right_table>
(3) ON <join_condition>
(4) WHERE <where_condition>
(5) GROUP BY <group_by_list>
(6) HAVING <having_condition>
(9) ORDER BY <order_by_condition>
(10) LIMIT <limit_number>

    FROM ..., ... -> ON -> (LEFT/RIGHT JOIN) -> WHERE ->
GROUP BY -> HAVING -> SELECT -> DISTINCT ->
ORDER BY -> LIMIT

    SELECT жҳҜе…Ҳжү§иЎҢ FROM иҝҷдёҖжӯҘзҡ„гҖӮеңЁиҝҷдёӘйҳ¶ж®өпјҢеҰӮжһңжҳҜеӨҡеј иЎЁиҒ”жҹҘпјҢиҝҳдјҡз»ҸеҺҶдёӢйқўзҡ„еҮ дёӘжӯҘйӘӨ:

    1. йҰ–е…Ҳе…ҲйҖҡиҝҮ CROSS JOIN жұӮз¬ӣеҚЎе°”з§ҜпјҢзӣёеҪ“дәҺеҫ—еҲ°иҷҡжӢҹиЎЁ vt(virtual table)1-1
    2. йҖҡиҝҮ ON иҝӣиЎҢзӯӣйҖүпјҢеңЁиҷҡжӢҹиЎЁ vt1-1 зҡ„еҹәзЎҖдёҠиҝӣиЎҢзӯӣйҖүпјҢеҫ—еҲ°иҷҡжӢҹиЎЁ vt1-2
    3. ж·»еҠ еӨ–йғЁиЎҢгҖӮеҰӮжһңжҲ‘д»¬дҪҝз”Ёзҡ„жҳҜе·ҰиҝһжҺҘгҖҒеҸій“ҫжҺҘжҲ–иҖ…е…ЁиҝһжҺҘпјҢе°ұдјҡж¶үеҸҠеҲ°еӨ–йғЁиЎҢпјҢд№ҹе°ұжҳҜеңЁиҷҡжӢҹиЎЁ vt1-2 зҡ„еҹәзЎҖдёҠеўһеҠ еӨ–йғЁиЎҢпјҢеҫ—еҲ°иҷҡжӢҹиЎЁ vt1-3гҖӮ

    еҪ“з„¶еҰӮжһңжҲ‘д»¬ж“ҚдҪңзҡ„жҳҜдёӨеј д»ҘдёҠзҡ„иЎЁпјҢиҝҳдјҡйҮҚеӨҚдёҠйқўзҡ„жӯҘйӘӨпјҢзӣҙеҲ°жүҖжңүиЎЁйғҪиў«еӨ„зҗҶе®ҢдёәжӯўгҖӮиҝҷдёӘиҝҮзЁӢеҫ—еҲ°жҳҜжҲ‘д»¬зҡ„еҺҹе§Ӣж•°жҚ®гҖӮ

    еҪ“жҲ‘д»¬жӢҝеҲ°дәҶжҹҘиҜўж•°жҚ®иЎЁзҡ„еҺҹе§Ӣж•°жҚ®пјҢд№ҹе°ұжҳҜжңҖз»Ҳзҡ„иҷҡжӢҹиЎЁ vt1 пјҢе°ұеҸҜд»ҘеңЁжӯӨеҹәзЎҖдёҠеҶҚиҝӣиЎҢ WHERE йҳ¶ж®өгҖӮеңЁиҝҷдёӘйҳ¶ж®өдёӯпјҢдјҡж №жҚ® vt1 иЎЁзҡ„з»“жһңиҝӣиЎҢзӯӣйҖүиҝҮж»ӨпјҢеҫ—еҲ°иҷҡжӢҹиЎЁ vt2 гҖӮ

    з„¶еҗҺиҝӣе…Ҙз¬¬дёүжӯҘе’Ңз¬¬еӣӣжӯҘпјҢд№ҹе°ұжҳҜ GROUP BY е’Ң HAVING йҳ¶ж®өгҖӮеңЁиҝҷдёӘйҳ¶ж®өдёӯпјҢе®һйҷ…дёҠжҳҜеңЁиҷҡжӢҹиЎЁ vt2 зҡ„еҹәзЎҖдёҠиҝӣиЎҢеҲҶз»„е’ҢеҲҶз»„иҝҮж»ӨпјҢеҫ—еҲ°дёӯй—ҙзҡ„иҷҡжӢҹиЎЁ vt3 е’Ң vt4 гҖӮ

    еҪ“жҲ‘д»¬е®ҢжҲҗдәҶжқЎд»¶зӯӣйҖүйғЁеҲҶд№ӢеҗҺпјҢе°ұеҸҜд»ҘзӯӣйҖүиЎЁдёӯжҸҗеҸ–зҡ„еӯ—ж®өпјҢд№ҹе°ұжҳҜиҝӣе…ҘеҲ° SELECT е’Ң DISTINCT йҳ¶ж®өгҖӮ

    йҰ–е…ҲеңЁ SELECT йҳ¶ж®өдјҡжҸҗеҸ–жғіиҰҒзҡ„еӯ—ж®өпјҢз„¶еҗҺеңЁ DISTINCT йҳ¶ж®өиҝҮж»ӨжҺүйҮҚеӨҚзҡ„иЎҢпјҢеҲҶеҲ«еҫ—еҲ°дёӯй—ҙзҡ„иҷҡжӢҹиЎЁ vt5-1 е’Ң vt5-2гҖӮ

    еҪ“жҲ‘д»¬жҸҗеҸ–дәҶжғіиҰҒзҡ„еӯ—ж®өж•°жҚ®д№ӢеҗҺпјҢе°ұеҸҜд»ҘжҢүз…§жҢҮе®ҡзҡ„еӯ—ж®өиҝӣиЎҢжҺ’еәҸпјҢд№ҹе°ұжҳҜ ORDER BY йҳ¶ж®өпјҢеҫ—еҲ°иҷҡжӢҹиЎЁ vt6 гҖӮ

    жңҖеҗҺеңЁ vt6 зҡ„еҹәзЎҖдёҠпјҢеҸ–еҮәжҢҮе®ҡиЎҢзҡ„и®°еҪ•пјҢд№ҹе°ұжҳҜ LIMIT йҳ¶ж®өпјҢеҫ—еҲ°жңҖз»Ҳзҡ„з»“жһңпјҢеҜ№еә”зҡ„жҳҜиҷҡжӢҹиЎЁ vt7 гҖӮ

    Hive SQL иҜӯеҸҘжү§иЎҢйЎәеәҸ

    ж ҮеҮҶзҡ„ Hive SQL жҹҘиҜўиҜӯеҸҘдёҺ MySQL зҡ„зӣёеҗҢпјҢдҪҶжҳҜз”ұдәҺдёӨиҖ…еә•еұӮзҡ„иҝҗиЎҢеҺҹзҗҶдёҚдёҖж ·пјҢжҹҘиҜўиҜӯеҸҘзҡ„еә•еұӮжү§иЎҢйЎәеәҸз•ҘжңүдёҚеҗҢпјҢжҲ‘д»¬еҸҜд»ҘжӢҶеҲҶжҲҗ map йҳ¶ж®өе’Ң reduce йҳ¶ж®өжқҘзңӢгҖӮ

    Map йҳ¶ж®өпјҡ

    1. жү§иЎҢ from еҠ иҪҪпјҡиҝӣиЎҢиЎЁзҡ„жҹҘжүҫдёҺеҠ иҪҪгҖӮ
    2. жү§иЎҢ where иҝҮж»ӨпјҡиҝӣиЎҢжқЎд»¶иҝҮж»ӨдёҺзӯӣйҖүгҖӮ
    3. жү§иЎҢ select жҹҘиҜўпјҡиҝӣиЎҢиҫ“еҮәйЎ№зҡ„зӯӣйҖүгҖӮ
    4. жү§иЎҢ group by еҲҶз»„пјҡеҜ№ж•°жҚ®иҝӣиЎҢеҲҶз»„гҖӮ
    5. map з«Ҝж–Үд»¶еҗҲе№¶пјҡmap з«Ҝжң¬ең°жәўеҮәеҶҷж–Үд»¶зҡ„еҗҲе№¶ж“ҚдҪңпјҢжҜҸдёӘ map жңҖз»ҲеҪўжҲҗдёҖдёӘдёҙж—¶ж–Үд»¶гҖӮз„¶еҗҺжҢүеҲ—жҳ е°„еҲ°еҜ№еә”зҡ„ reduce йҳ¶ж®өгҖӮ

    Reduce йҳ¶ж®өпјҡ

    1. group byпјҡеҜ№ map з«ҜеҸ‘йҖҒиҝҮжқҘзҡ„ж•°жҚ®иҝӣиЎҢеҲҶз»„е№¶иҝӣиЎҢи®Ўз®—гҖӮ
    2. selectпјҡжңҖеҗҺиҝҮж»ӨеҲ—з”ЁдәҺиҫ“еҮәз»“жһңгҖӮ
    3. limit жҺ’еәҸеҗҺиҝӣиЎҢз»“жһңиҫ“еҮәеҲ° HDFS ж–Үд»¶гҖӮ

    Hive SQL иҜӯеҸҘзҡ„жү§иЎҢеҺҹзҗҶ

    Join зҡ„е®һзҺ°еҺҹзҗҶ

    select u.name, o.orderid from order o join user u on o.uid = u.uid;

    еңЁ map зҡ„иҫ“еҮә value дёӯдёәдёҚеҗҢиЎЁзҡ„ж•°жҚ®жү“дёҠ tag ж Үи®°пјҢеңЁ reduce йҳ¶ж®өж №жҚ® tag еҲӨж–ӯж•°жҚ®жқҘжәҗгҖӮ

    Group By зҡ„е®һзҺ°еҺҹзҗҶ

    select rank, isonline, count(*) from city group by rank, isonline;

    е°Ҷ Group By зҡ„еӯ—ж®өз»„еҗҲдёә map зҡ„иҫ“еҮә key еҖјпјҢеҲ©з”Ё MapReduce зҡ„жҺ’еәҸпјҢеңЁ reduce йҳ¶ж®өдҝқеӯҳ LastKey еҢәеҲҶдёҚеҗҢзҡ„ keyгҖӮ

    Distinct зҡ„е®һзҺ°еҺҹзҗҶ

    select dealid, count(distinct uid) num from order group by dealid;

    еҪ“еҸӘжңүдёҖдёӘ distinct еӯ—ж®өж—¶пјҢеҰӮжһңдёҚиҖғиҷ‘ Map йҳ¶ж®өзҡ„ Hash GroupByпјҢеҸӘйңҖиҰҒе°Ҷ Group By еӯ—ж®өе’Ң Distinct еӯ—ж®өз»„еҗҲдёә map иҫ“еҮә keyпјҢеҲ©з”Ё mapreduce зҡ„жҺ’еәҸпјҢеҗҢж—¶е°Ҷ Group By еӯ—ж®өдҪңдёә reduce зҡ„ keyпјҢеңЁ reduce йҳ¶ж®өдҝқеӯҳ LastKey еҚіеҸҜе®ҢжҲҗеҺ»йҮҚгҖӮ
